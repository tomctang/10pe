#112 Bouncy Numbers
from time import time
start=time()
def listint(num):
    integer=''
    for number in num:
        integer=integer+str(number)
    return int(integer)
def bouncy(n):
    input=list(map(int, str(n)))
    output=input[:]
    output.sort()
    if input==output:
        return False
    output.reverse()
    if input==output:
        return False
    return True
def bouncy_percentage(n):
    sum_bouncy=0
    for i in range(1, n+1):
        if bouncy(i)==True:
            sum_bouncy+=1
    ratio=sum_bouncy/n
    return ratio*100
print(bouncy_percentage(1587000))
end=time()
print(end-start)

# 124 Ordered Radicals
# find quicker solution
def prime_factors(n):
    factors=[]
    for i in range(2,n):
        if prime(i) and n%i==0:
            factors.append(i)
    return factors

def rad(n):
    if n==1 or prime(n):
        return n
    fac=prime_factors(n)
    ans=fac[0]
    for i in range(1,len(fac)):
        ans*=fac[i]
    return ans

for i in range(2,3000):
    rad(i)
