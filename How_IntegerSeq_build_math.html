# 0) Simple recurrence (baseline)
def fib_rec(n):
    if n < 0:
        return 0
    a,b = 0,1
    for _ in range(n):
        a,b = b,a+b
    return a

# 1) 2x2 Matrix power (Binet via eigenvalues)
Q = Matrix([[1,1],[1,0]])

def fib_matrix(n):
    if n==0:
        return 0
    M = Q**(n-1)   # Q^{n-1} * [F1; F0] = [Fn; F_{n-1}]
    return M[0,0]

# 2) Generating function viewpoint (algebraic OGF and EGF ODE)
# Ordinary generating function G(x) = x/(1 - x - x^2) satisfies algebraic equation
# (1 - x - x^2) * G(x) = x
# Exponential generating function E(x) = sum_{n>=0} F_n x^n / n! satisfies a linear ODE with lower-order correction terms:
# E'' - E' - E = c0 + c1*x^{-1} terms coming from initial indices; we verify numerically for coefficients.

def ogf_series(N=20):
    x = var('x')
    G = x/(1 - x - x^2)
    return G.series(x, N)

# Verify EGF differential relation numerically by power series

def egf_series_check(N=12):
    x = var('x')
    # build EGF as formal series using fib_rec
    E = sum(fib_rec(n)*x**n/factorial(n) for n in range(N))
    L = derivative(E, x, 2) - derivative(E, x, 1) - E
    return (E.series(x, N), L.series(x, N))

# 3) Combinatorial / tilings (group action & tiling)
# Number of ways to tile a 1 x n board with 1x1 and 1x2 tiles equals F_{n+1}

def tilings_count(n):
    # simple DP: ways[n] = ways[n-1] + ways[n-2]
    ways = [1,1]  # ways[0]=1 (empty), ways[1]=1
    for k in range(2, n+1):
        ways.append(ways[-1] + ways[-2])
    return ways[n]

# 4) Graph-theoretic: number of independent sets in path graph P_n equals F_{n+2}

def indep_sets_path(n):
    # DP: a_n = a_{n-1} + a_{n-2} with initial a_0 = 1, a_1 = 2
    a = [1,2]
    for k in range(2, n+1):
        a.append(a[-1] + a[-2])
    return a[n]

# 5) Dirichlet / zeta-style generating function (numerical exploration)
def dirichlet_F(s, terms=2000):
    # D(s) = sum_{n>=1} F_n / n^s
    return sum(Integer(fib_rec(n))*n**(-s) for n in range(1, terms))

# 6) Lambert series exploration: L(q) = sum_{n>=1} F_n q^n / (1 - q^n)
# numeric series evaluation for |q|<1

def lambert_series(q, terms=200):
    q = QQ(q)
    S = 0
    for n in range(1, terms):
        S += fib_rec(n)*q**n/(1 - q**n)
    return S

# 7) Temperley-Lieb / anyon-style fusion spaces (representation-theoretic viewpoint)
# Fibonacci anyon fusion dimensions satisfy the same recurrence:
# dim(n) = dim(n-1) + dim(n-2), with dim(0)=1, dim(1)=1 (or other base depending on boundary conditions)

def fusion_dim(n):
    return fib_rec(n+1)  # depending on indexing conventions

# 8) Elliptic-curve / Frobenius eigenvalue toy: Lucas sequences from trace of Frobenius
# This is a short demonstration: many linear recurrences (including Fibonacci/Lucas) arise from traces of Frobenius
# for rank-1 motives. We show a small example computing #E(F_p) for a curve and forming a_n = p + 1 - #E(F_p).

        seq.append(ap)

def count_no_consecutive_b(n, memo={0: 1, 1: 2}):
    """Count binary strings of length n with no consecutive 1's (or b's)"""
    if n in memo:

    # Either ends in 'a' (0) or 'b' (1)

    result = count_no_consecutive_b(n-1, memo) + count_no_consecutive_b(n-2, memo)

    return result

def count_dyck_paths(n):
    """Count Dyck paths using dynamic programming"""
    if n == 0:
        return 1
    # dp[i][j] = number of paths of length i ending at height j
    dp = [[0]*(n+1) for _ in range(2*n+1)]
    dp[0][0] = 1
    
    for i in range(2*n):
        for j in range(n+1):
            if dp[i][j] > 0:
                # Can go up if not at max height
                if j < n:
                    dp[i+1][j+1] += dp[i][j]
                # Can go down if not at ground
                if j > 0:
                    dp[i+1][j-1] += dp[i][j]
    
    return dp[2*n][0]

def partition_euler(n, memo={0: 1}):
    if n < 0:
        return 0
    if n in memo:
        return memo[n]
    
    result = 0
    k = 1
    while True:
        pent1 = k*(3*k - 1)//2
        pent2 = k*(3*k + 1)//2
        
        if pent1 > n:
            break
            
        sign = (-1)^(k+1)
        result += sign * partition_euler(n - pent1, memo)
        
        if pent2 <= n:
            result += sign * partition_euler(n - pent2, memo)
        
        k += 1
    
    memo[n] = result
    return result
