def solve(days,n):
    if days==['M']*n:
        return n
    max_money=n
    for day in range(n):
        if days[day]=='C':
            s=1
            money=0
            for i in range(day+1):
                if days[i]=='C':
                    s+=1
                else:

            money+=s*(n-day-1)
