#92 Square Digit Chains
def chain(n):
    while n!=1 and n!=89:
        num=list(map(int, str(n)))
        squared=0
        for digit in num:
            squared+=digit**2
        n=squared
    return n
num_89=0
for i in range(1, 10000000):
    if chain(i)==89:
        num_89+=1
    if i%50000==0:
        print(i)
