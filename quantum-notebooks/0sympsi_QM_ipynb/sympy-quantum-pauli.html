from IPython.display import HTML

def print_table(data):
    t_table = "<table>\n%s\n</table>"
    t_row = "<tr>%s</tr>"
    t_col = "<td>%s</td>"
    table_code = t_table % "".join([t_row % "".join([t_col % ("$%s$" % latex(col))
                                                     for col in row])
                                    for row in data])
    return HTML(table_code)
